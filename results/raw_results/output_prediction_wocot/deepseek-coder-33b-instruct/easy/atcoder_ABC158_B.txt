def solve(n, a, b):
    if a > b:
        a, b = b, a
    if a == b:
        return a
    if a == 0:
        return 0
    if a == 1:
        return b
    if b % a == 0:
        return a
    return b // a * a

n, a, b = map(int, input().split())
print(solve(n, a, b))
