        print(a, end=' ')
        a, b = b, a+b
    print()

def is_prime(n):
    if n <= 1:
        return False
    if n <= 3:
        return True
    if n % 2 == 0 or n % 3 == 0:
        return False
    i = 5
    while i * i <= n:
        if n % i == 0 or n % (i + 2) == 0:
            return False
        i += 6
    return True

def prime_factors(n):
    factors = []
    while n % 2 == 0:
        factors.append(2),
        n = n / 2
    for i in range(3,int(n**0.5)+1,2):
        while n % i== 0:
            if is_prime(i):
                factors.append(i)
            n = n / i
    if n > 2:
        factors.append(n)
    return factors

print(prime_factors(315))

def is_pangram(s):
    alphabet = "abcdefghijklmnopqrstuvwxyz"
    for char in alphabet:
        if char not in s.lower():
            return False
    return True

print(is_pangram('the quick brown fox jumps over the lazy dog'))

def is_armstrong(n):
    sum = 0
    temp = n
    while temp > 0:
        digit = temp % 10
        sum += digit ** 3
        temp //= 10
    return n == sum

print(is_armstrong(153))
